C) Map marker location - top

To change the marker location or the image on the map for the contact page all you need to do is edit line 43 in custom.js

		  	if ($("#map").length > 0){
				$('#map').gmap3({
					map: {
					    options:{
					        zoom:16, /** Set your zoom **/
					        center: [51.576084, 0.488736], /** Enter your long/lat **/
					        mapTypeId: google.maps.MapTypeId.MAP,
					        mapTypeControl: false,
					        mapTypeControlOptions: {
					          style: google.maps.MapTypeControlStyle.DROPDOWN_MENU
					        },
					        navigationControl: false,
					        scrollwheel: false,
					        streetViewControl: false
					    }
					},
					marker:{
					    latLng: [51.576084, 0.488736], /** Enter your marker long/lat **/
					    options: {
						    icon: new google.maps.MarkerImage(
						        "images/map-pin.png", new google.maps.Size(223, 167, "px", "px") /** Enter your pin location **/
						    )
					    }
					 }
					}
			
				);
			}
